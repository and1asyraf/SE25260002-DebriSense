DATABASE DESIGN MERMAID 

```mermaid
erDiagram
    USER ||--o{ RIVER : creates
    USER ||--o{ HOTSPOT_REPORT : submits
    USER ||--o{ WATCHLIST : manages
    USER ||--o{ LOCATION_REQUEST : submits
    USER ||--o{ ALERT : receives
    RIVER ||--o{ DRI_READING : generates
    RIVER ||--o{ HOTSPOT_REPORT : receives
    RIVER ||--o{ WATCHLIST : tracked_in
    RIVER ||--o{ ALERT : triggers

    USER {
        int id PK
        string email
        string password_hash
        string ngo_name
        string contact_person
        string phone
        string role
        boolean is_verified
        datetime created_at
        datetime last_login
    }

    RIVER {
        int id PK
        string name
        float latitude
        float longitude
        string info
        float baseline_plastic
        float baseline_organic
        float baseline_household
        float baseline_industrial
        float baseline_others
        datetime created_at
        int created_by FK
    }

    DRI_READING {
        int id PK
        int river_id FK
        string risk_level
        float rainfall_level
        float wind_speed
        float tide_level
        float water_flow
        float current_dri
        float baseline_dri
        float debris_estimate_kg
        datetime timestamp
        string land_use
    }

    HOTSPOT_REPORT {
        int id PK
        int river_id FK
        int reporter_id FK
        string status
        float actual_payload
        string debris_type_breakdown
        date sighting_date
        datetime reported_at
    }

    WATCHLIST {
        int id PK
        int user_id FK
        int river_id FK
        boolean alert_on_high
        boolean alert_on_critical
        boolean email_alerts
        datetime added_at
    }

    LOCATION_REQUEST {
        int id PK
        int user_id FK
        string name
        float latitude
        float longitude
        string reason
        string status
        datetime submitted_at
        datetime reviewed_at
    }

    ALERT {
        int id PK
        int user_id FK
        int river_id FK
        string title
        string message
        boolean is_read
        datetime created_at
    }
```
